function hangingEdges(DirectedGraph, edges) {
    'use strict';
    var err = [],
        from,
        to;
    edges.forEach(function (edge) {
        var from = DirectedGraph.getNode(edge.from),
            to = DirectedGraph.getNode(edge.to);
        if (!from || !to) {
            err.push(edge);
        }
    });
    return err;
}

function DirectedGraph(struct) {
